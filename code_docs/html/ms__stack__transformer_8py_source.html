101 def _load_stack_args(args, stack):
102 
103  # Process each argument.
104  while (len(args) > 0):
105 
106  # Pull an arg.
107  item = args[0]
108  args = args[1:]
109 
110  # If this is an XLM object just push the stack of the XLM object.
111  if (isinstance(item, XLM_Object)):
112  for s in item.stack:
113  stack.append(s)
114  continue
115 
116  # Simple case is a literal.
117  if (not isinstance(item, list)):
118  stack.append(item)
119  continue
120 
121  # Single item as 2nd argument to an infix operator.
122  if (len(item) == 1):
123  stack.append(item[0])
124  continue
125 
126  # 2nd item (infix) in the list is the operator.
127  op = item[1]
128  first_arg = item[0]
129  second_arg = item[2:]
130  if (len(second_arg) == 1):
131  second_arg = second_arg[0]
132 
133  # Nested expression involving another operator with args.
134  stack = _load_stack([op, [first_arg, second_arg]], stack)
135 
136  # Return the updated stack.
137  return stack

149 def _load_stack(items, stack):
