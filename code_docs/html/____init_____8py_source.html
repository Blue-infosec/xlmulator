60 def _extract_xlm(maldoc):
61 
62  # Run olevba on the given file.
63  olevba_out = None
64  FNULL = open(os.devnull, 'w')
65  try:
66  cmd = "timeout 30 olevba -c \"" + str(maldoc) + "\""
67  olevba_out = subprocess.check_output(cmd, shell=True, stderr=FNULL)
68  except Exception as e:
69  color_print.output('r', "ERROR: Running olevba on " + str(maldoc) + " failed. " + str(e))
70  return None
71 
72  # Not handling encrypted Excel files.
73  if (b"FILEPASS record: file is password protected" in olevba_out):
74  color_print.output('y', "WARNING: " + str(maldoc) + " is password protected. Not emulating.")
75  return None
76 
77  # Pull out the chunks containing the XLM lines.
78  chunk_pat = b"in file: xlm_macro \- OLE stream: 'xlm_macro'\n(?:\- ){39}\n(.+)"
79  chunks = re.findall(chunk_pat, olevba_out, re.DOTALL)
80 
81  # Pull out all the XLM lines from each chunk.
82  r = b""
83  xlm_pat = br"' \d\d\d\d {1,10}\d{1,6} [^\n]+\n"
84  for chunk in chunks:
85  for line in re.findall(xlm_pat, chunk):
86  r += line
87 
88  # Convert characters so this can be parsed.
89  try:
90  r = XLM.utils.to_str(r)
91  except UnicodeDecodeError:
92  r = XLM.utils.strip_unprintable(r)
93 
94  # Did we find XLM?
95  if (len(r.strip()) == 0):
96  color_print.output('y', "WARNING: No XLM found.")
97  return None
98 
99  # Done. Return XLM lines.
100  return r

111 def _guess_xlm_sheet(workbook):
112 
113  # TODO: If plugin_biff.py used by olevba to dump XLM includes sheet names this
114  # function will no longer be needed.
115 
116  # Look at each sheet.
117  xlm_sheet = None
118  unresolved_count = -1
119  for curr_sheet_name in workbook.sheet_names():
120  curr_sheet = workbook.sheet_by_name(curr_sheet_name)
121  curr_unresolved_count = 0
122  for cell_value in list(curr_sheet.cells.values()):
123  cell_value = cell_value.strip()
124  if (len(cell_value) == 0):
125  continue
126  if (cell_value.strip() == "#NAME?"):
127  curr_unresolved_count += 1
128  if (curr_unresolved_count > unresolved_count):
129  unresolved_count = curr_unresolved_count
130  xlm_sheet = curr_sheet_name
131 
132  # Return the sheet with the most '#NAME' cells.
133  return xlm_sheet

224  print("=========== DONE 2007+ CONTENTS ==============")

227  xlm_sheet_name = None
228  max_formulas = -1
229  for sheet in workbook_info.keys():
230  if (len(workbook_info[sheet]) > max_formulas):

237

240  col = cell_index[1]
241  if (row not in xlm_cells):
242  xlm_cells[row] = {}
243  raw_formula = workbook_info[xlm_sheet_name][cell_index][0]
244  if (raw_formula is None):
245 
246  # Do we have a value?
247  formula_val = workbook_info[xlm_sheet_name][cell_index][1]
248  if (formula_val is not None):
249 
250  # Just save the value in the cell.
251  xlm_cells[row][col] = formula_val
252  continue

273  return (None, None, None)
