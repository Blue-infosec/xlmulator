45 def fix_olevba_xlm(xlm_code):
46 
47  # plugin_biff does not escape newlines in strings. Try to find them and fix them.
48  xlm_code = xlm_code.strip().encode()
49  r = b""
50  mod_chunk = b""
51  lines = xlm_code.split(b"\n")
52  pos = -1
53  new_line = b""
54  xlm_pat = br"' \d\d\d\d {1,10}\d{1,6} [^\n]+\n"
55  while (pos < (len(lines) - 1)):
56 
57  # Start putting together an aggregated line?
58  pos += 1
59  curr_line = lines[pos]
60  if (curr_line.startswith(b"' ")):
61  mod_chunk += new_line + b"\n"
62  new_line = curr_line
63  continue
64 
65  # This line is part of a string with unescaped newlines.
66  new_line += b"\\n" + curr_line
67  if (len(new_line) > 0):
68  mod_chunk += b"\n" + new_line
69  mod_chunk = mod_chunk.strip() + b"\n"
70 
71  # Handle double quotes in strings.
72  for line in re.findall(xlm_pat, mod_chunk):
73 
74  # plugin_biff does not escape double quotes in strings. Try to find them
75  # and fix them.
76  #
77  # ' 0006 72 FORMULA : Cell Formula - R9C1 len=50 ptgRefV R7C49153 ptgStr "Set wsh = CreateObject("WScript.Shell")" ptgFuncV FWRITELN (0x0089)
78  str_pat = b"Str \".*?\" ptg"
79  str_pat1 = b"Str \"(.*?)\" ptg"
80  for old_str in re.findall(str_pat, line):
81  tmp_str = re.findall(str_pat1, old_str)[0]
82  if (b'"' in tmp_str):
83  # Escape single quotes.
84  escaped_str = old_str[5:-5].replace(b"'", b"&apos;")
85  new_str = b"Str '" + escaped_str + b"' ptg"
86 
87  line = line.replace(old_str, new_str)
88  r += line
89 
90  # Done.
91  return r

103 def parse_olevba_xlm(xlm_code):
104 
105  # Fix some escaping issues before parsing.
106  xlm_code = XLM.utils.to_str(fix_olevba_xlm(xlm_code))

108  # Parse the olevba XLM.

110  try:
111  xlm_parser = Lark(xlm_grammar, start="lines", parser='lalr')
112  xlm_ast = xlm_parser.parse(xlm_code)
113  except UnexpectedInput as e:
114  XLM.color_print.output('r', "ERROR: Parsing olevba XLM failed.\n" + str(e))
115  num_pat = r"at line (\d+), column (\d+)"
116  error_loc = re.findall(num_pat, str(e))[0]
117  line_num = int(error_loc[0]) - 1
118  col_num = int(error_loc[1]) - 1
119  lines = xlm_code.split("\n")
120  bad_line = lines[line_num]
121  bad_line = bad_line[:col_num] + "...ERROR START...-->" + bad_line[col_num:]
122  XLM.color_print.output('r', "BAD LINE: " + bad_line)
123  return None
