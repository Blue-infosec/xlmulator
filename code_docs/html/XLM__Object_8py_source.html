392  raise ValueError("The stack is None.")

400  # If this is not a function there is nothing to do.

402 
403  # Just convert the top stack item to a string and we are done.
404  r = str(curr_item)
405  return (r, tmp_stack)
406 
407  # We have a function.
408 
409  # Infix function? These always have 2 arguments.
410  if (curr_item.is_infix_function()):
411

414  raise ValueError("Infix operator '" + str(curr_item) + "' requires 2 arguments.")
415

417  arg2, tmp_stack = _get_str(tmp_stack)
418  arg1, tmp_stack = _get_str(tmp_stack)
419

421  r = str(arg1) + str(curr_item) + str(arg2)
422  return (r, tmp_stack)
423 
424  # Non-infix function. These have a variable # of arguments.
425  num_args = curr_item.get_num_args()
426 
427  # Sanity check.
428  if (len(tmp_stack) < num_args):
429  print(tmp_stack)
430  raise ValueError("Operator '" + str(curr_item) + "' requires " + str(num_args) + " arguments.")
431 
432  # Resolve the strings for all the arguments.
433  first = True
434  args = ""
435  for i in range(0,num_args):
436  if (not first):
437  args = "," + args
438  first = False
439  arg, tmp_stack = _get_str(tmp_stack)
440  args = str(arg) + args
441 
442  # Return the string for the function.
443  r = str(curr_item) + "(" + args + ")"
