726 def eval(func_name, params, sheet):
727 
728  # Do we know how to emulate this?
729  if (func_name not in func_lookup.keys()):
730  raise ValueError("Do not know how to emulate '" + str(func_name) + "'.")
731 
732  # Get the emulator for the function and run it.
733  func = func_lookup[func_name]
734 
735  # FORMULA can compute a value and update a cell with that computed value. Track
736  # the cell to update in this case.
737  update_index = None
738  if ((func_name == "FORMULA") and (len(params) > 1)):
739  # $R54$C54
740  update_fields = str(params[1]).replace("$C", ":").replace("$R", "").split(":")
741  update_index = (int(update_fields[0]), int(update_fields[1]))
742 
743  # Evaluate the parameters.
744  eval_params = []
745  for p in params:
746  if (hasattr(p, "eval")):
747  eval_params.append(p.eval(sheet))
748  else:
749  eval_params.append(p)
750  r = func(eval_params, sheet)
751 
752  # Does this value get written to a cell?
